;; Basic Dynamic Binding - Value Comparison
;; Run in Clojure: Each expression should be wrapped in (println ...)
;; Run in our impl: Just evaluate each expression

(def ^:dynamic *x* 10)
*x*

(binding [*x* 20] *x*)

*x*

(def ^:dynamic *y* 100)
(binding [*x* 1 *y* 2] (+ *x* *y*))

(binding [*x* 1]
  (binding [*x* 2]
    (binding [*x* 3] *x*)))

*x*

(binding [*x* 10]
  (binding [*x* 20] *x*))

(binding [*x* 5] (* *x* *x*))

(binding [*x* 1]
  (+ *x*
     (binding [*x* 2]
       (+ *x*
          (binding [*x* 3] *x*)))))

*x*

:quit
