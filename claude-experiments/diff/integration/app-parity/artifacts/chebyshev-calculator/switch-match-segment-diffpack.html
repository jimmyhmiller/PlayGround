#include <stdio.h> // Evaluates a Chebyshev expansion at a given // x value using the Clenshaw algorithm. float chebyshevEval(const float* coeffs, int num_coeffs, float x, float x_min, float x_max) { float x_rel_2 = -2.0 + 4.0 * (x - x_min) / (x_max - x_min); float d = 0.0; float dd = 0.0; float temp = 0.0; for (int i = num_coeffs - 1; i > 0; i--) { temp = d; d = x_rel_2 * d - dd + coeffs[i]; dd = temp; } return 0.5 * x_rel_2 * d - dd + 0.5 * coeffs[0]; } // 12 term expansion coefficients for // f(x)=Math.exp(-0.4 * x) * Math.sin(x * x) // x_min=0, x_max=3.65 #define NUM_COEFFS 12 float coeffs[NUM_COEFFS] = { 0.19327931893427666, -0.11012678669940786, -0.07529132651642957, 0.1969526660161555, -0.05711544015934935, -0.07083056543988137, 0.19207939848880035, 0.14021901363038325, -0.05751945836850155, -0.10023643897350684, -0.02732265178808006, 0.0154922925898354 }; float x_min = 0; float x_max = 3.65; int main() { // Evaluate the approximation at the interval midpoint float x_mid = 0.5 * (x_min + x_max); float value_at_x_mid = chebyshevEval(coeffs, NUM_COEFFS, x_mid, x_min, x_max); printf("Approximated value at x=%f is %f (single precision)\n", x_mid, value_at_x_mid); printf("Should be -0.10746065924500293 (double precision)"); return 0; }
